import pandas as pd
import statsmodels.formula.api as smf
from sklearn.linear_model import LinearRegression
from patsy import dmatrices
from sklearn.model_selection import train_test_split
from sklearn.cluster import KMeans
from sklearn.decomposition import PCA
import seaborn as sns
df=pd.read_csv("housing_renamed.csv")
def scores(formula):
    lr2=LinearRegression()
    response,predictors=dmatrices(formula,data=df)
    xtrain,xtest,ytrain,ytest = train_test_split(predictors,response,random_state=0)
    predicted2=lr2.fit(xtrain,ytrain)
    return predicted2.score(xtest,ytest)

2. 使用iris.csv数据进行聚类分析：不用variety列，使用前四列数据进行Kmeans聚类分析(k=3)，并使用pca降维后作图显示聚类结果，然后与variety进行比较。¶

In [ ]:
df2tot=pd.read_csv("iris(1).csv")
df2=df2tot.iloc[:,:4]
kmeans=KMeans(n_clusters=3,random_state=42,).fit(df2.values)
kmeanstmp=pd.DataFrame(kmeans.labels_,columns=["cluster"])
pca=PCA(n_components=2).fit(df2)
pca_trans=pca.transform(df2)
pcatrans_df=pd.DataFrame(pca_trans,columns=["pca1","pca2"])
kmeans3=pd.concat([kmeanstmp,pcatrans_df,df2tot["variety"]],axis=1)
sns.lmplot(x="pca1",y="pca2",data=kmeans3,fit_reg=False,row="cluster",col="variety")
sns.lmplot(x="pca1",y="pca2",data=kmeans3,hue="cluster",fit_reg=False)
sns.lmplot(x="pca1",y="pca2",data=kmeans3,hue="variety",fit_reg=False)
